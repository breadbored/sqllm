--- AGENT ---

This query denormalizes and enriches raw video playback event data by extracting nested JSON fields, handling missing values through fallback logic, and computing temporal relationships between events.

**Step 1 — best_placement_query CTE:** Aggregates events where video is "playing", grouping by a composite unique identifier (videoId + uniqueId combination), domain, and URL path. Counts distinct sub_sessions for each combination to measure prevalence.

**Step 2 — best_placement CTE:** For each unique video+ID pair, uses MAX_BY to identify the domain and URL path associated with the most sessions (the modal/most common values), storing this as a lookup table.

**Step 3 — Main query:** Selects all base event columns except the JSON metadata blob, then:
- Extracts individual fields from action_meta JSON (user identifier, player settings like captions/fullscreen/volume, playback state like progress/duration/playing status, URL components, and UTM parameters)
- Applies three-tier fallback logic for missing values: use the actual field if present, else use the best_placement modal domain/path, else use FIRST_VALUE window function to propagate the first non-null value within the session partition
- Handles date-dependent action_meta reconstruction: for events before 2026-01-01, reconstructs the JSON with computed domain and path values via JSON_SET
- Computes temporal playback metrics: captures the previous playing state via LAG, and calculates time until the next playing event via LEAD (populated only if the gap is under 45 seconds)
- Left joins with best_placement to attach modal domain and path for each video

The output is a flattened, gap-filled event stream suitable for analytical queries on video playback behavior.

```sql
WITH
best_placement_query AS (
  SELECT
    CONCAT(JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), '-', JSON_EXTRACT_SCALAR(e.action_meta, '$.uniqueId')) as unique_id,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.domain') as url_domain,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path') as url_path,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.uniqueId') as video_id,
    COUNT(DISTINCT e.sub_session_id) as sessions
  FROM
    ${ref("pre_vaas_events_page_visits")} as e
  WHERE
    JSON_EXTRACT_SCALAR(e.action_meta, '$.playing') = 'playing'
  AND
    JSON_EXTRACT_SCALAR(e.action_meta, '$.domain') IS NOT NULL
  AND
    JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path') IS NOT NULL
  GROUP BY
    unique_id,
    url_domain,
    url_path,
    video_id
),
best_placement AS (
  SELECT
    bp.unique_id as unique_id,
    MAX_BY(bp.url_path, bp.sessions) as best_path,
    MAX_BY(bp.url_domain, bp.sessions) as best_domain,
    MAX(bp.sessions) as score
  FROM
    best_placement_query as bp
  GROUP BY
    bp.unique_id
)
SELECT
  e.* EXCEPT(
    action_meta
  ),
  -- Add app specific columns below here from the metadata

  -- user
  CASE 
    -- INT ID
    WHEN SAFE_CAST(JSON_EXTRACT_SCALAR(e.action_meta, '$.externalId') AS INT64) > 0
      THEN
        JSON_EXTRACT_SCALAR(e.action_meta, '$.externalId')
    -- UUID
    WHEN REGEXP_CONTAINS(JSON_EXTRACT_SCALAR(e.action_meta, '$.externalId'), r'^[0-9a-f]{8}-[0-9a-f]{4}-[1-5][0-9a-f]{3}-[89abAB][0-9a-f]{3}-[0-9a-f]{12}$')
      THEN
        JSON_EXTRACT_SCALAR(e.action_meta, '$.externalId')
    -- Other ID types
    ELSE
      NULL
  END as external_user_id,
  -- settings
  JSON_EXTRACT_SCALAR(e.action_meta, '$.caption') as vaas_settings_captions,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.fullscreen') as vaas_settings_fullscreen,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.muted') as vaas_settings_muted,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.picture_in_picture') as vaas_settings_picture_in_picture,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.playbackSpeed') as vaas_settings_playback_speed,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.quality') as vaas_settings_quality,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.volume') as vaas_settings_volume,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.width') as vaas_settings_width,
  -- state
  JSON_EXTRACT_SCALAR(e.action_meta, '$.completed') as vaas_state_completed,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.duration') as vaas_state_duration,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.progress') as vaas_state_progress,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.percentage') as vaas_state_percentage,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.playing') as vaas_state_playing,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.visible') as vaas_state_visible,
  -- URL, path, and query params
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.url') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.url'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.url')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_url,
  COALESCE(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'),
    bp.best_path,
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_url_path,
  COALESCE(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.domain'),
    bp.best_domain,
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.domain')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_domain,

  IF(
    e.created >= '2026-01-01',
    PARSE_JSON(e.action_meta),
    JSON_SET(
        JSON_SET(
            PARSE_JSON(e.action_meta),
            '$.url_path',
            COALESCE(
                JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'),
                bp.best_path,
                FIRST_VALUE(
                JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path')
                ) OVER (
                PARTITION BY
                    e.sub_session_id,
                    e.vaas_page_visit
                ORDER BY
                    e.created ASC
                )
            )
        ),
        '$.domain',
        COALESCE(
            JSON_EXTRACT_SCALAR(e.action_meta, '$.domain'),
            bp.best_domain,
            FIRST_VALUE(
            JSON_EXTRACT_SCALAR(e.action_meta, '$.domain')
            ) OVER (
            PARTITION BY
                e.sub_session_id,
                e.vaas_page_visit
            ORDER BY
                e.created ASC
            )
        )
    )
  ) as action_meta,

  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.hash') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.hash'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.hash')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_hash,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_query,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.uniqueId') as vaas_url_unique_id,
  JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId') as vaas_url_video_id,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_campaign') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_campaign'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_campaign')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_utm_campaign,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_medium') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_medium'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_medium')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_utm_medium,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_term') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_term'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_term')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_utm_term,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_content') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_content'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_content')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_utm_content,
  IF(
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_source') IS NOT NULL,
    JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_source'),
    FIRST_VALUE(
      JSON_EXTRACT_SCALAR(e.action_meta, '$.searchParams.utm_source')
    ) OVER (
      PARTITION BY
        e.sub_session_id,
        e.vaas_page_visit
      ORDER BY
        e.created ASC
    )
  ) as vaas_url_utm_source,
  LAG(JSON_EXTRACT_SCALAR(e.action_meta, '$.playing')) OVER (PARTITION BY e.sub_session_id, JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), RTRIM(JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'), '/') ORDER BY e.created ASC) as last_vaas_state_playing,
  IF(
    LEAD(JSON_EXTRACT_SCALAR(e.action_meta, '$.playing')) OVER (PARTITION BY e.sub_session_id, JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), RTRIM(JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'), '/') ORDER BY e.created ASC) = 'playing'
  AND
    LEAD(e.seconds_from_last_event) OVER (PARTITION BY e.sub_session_id, JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), RTRIM(JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'), '/') ORDER BY e.created ASC) < 45,
    -- e.seconds_from_last_event,
    LEAD(e.seconds_from_last_event) OVER (PARTITION BY e.sub_session_id, JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), RTRIM(JSON_EXTRACT_SCALAR(e.action_meta, '$.url_path'), '/') ORDER BY e.created ASC),
    NULL
  ) as time_played,
FROM
  ${ref("pre_vaas_events_page_visits")} AS e
LEFT JOIN
  best_placement as bp
ON
  bp.unique_id = CONCAT(JSON_EXTRACT_SCALAR(e.action_meta, '$.videoId'), '-', JSON_EXTRACT_SCALAR(e.action_meta, '$.uniqueId'))
```
