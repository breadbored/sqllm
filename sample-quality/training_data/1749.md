Can you help me write a query that propagates the starting URL, referrer, and UTM parameters from the beginning of a session and subsession down to all subsequent events in that subsession?

--- AGENT ---

This query implements a URL and attribution parameter forwardfill pattern for web event data. It identifies the first occurrence of various attributes (URL path, referrer, UTM parameters) within a session and subsession boundary, then propagates those values to all subsequent events in that same subsession.

**Structure:**

The CTE `session_ids_with_start_urls` extracts the earliest (by event_num) URL, landing path, and UTM values for each session_id, establishing the session-level baseline.

The main query then:
1. Joins each event back to the `events_start_urls` table (which presumably contains subsession-level boundaries) using a lookback pattern: events are matched to a subsession start where the current event_num is greater than or equal to the start's event_num, within the same session, segment, and utm_visit group.
2. Uses COALESCE to select the subsession start URL if available, otherwise falls back to the event's own URL.
3. Attaches the session-level start URL separately.
4. For subsession-level UTM and referrer fields, uses conditional logic that outputs the subsession's starting values only when a change is detected (the start value differs from a "last" value) or when all UTM values are null. Otherwise it outputs NULL, indicating no attribution change occurred.

The output is one row per event (by `ne.id`) with session-level and subsession-level attribution attributes forwardfilled.

```sql
/**
* Start URL Forwardfill
*
* This layer is used to get the first URL, referrer, and UTMs in a subsession, and the first URL in a session.
**/
WITH
  -- Get the path a user went to as they started their first session
  session_ids_with_start_urls AS (
    SELECT
      _ne.session_id,
      MIN_BY (_ne.url_path, _ne.event_num) AS start_url_path,
      MIN_BY (_ne.landing_path, _ne.event_num) AS start_landing_url_path,
      MIN_BY (_ne.utm_campaign, _ne.event_num) AS start_utm_campaign,
      MIN_BY (_ne.utm_content, _ne.event_num) AS start_utm_content,
      MIN_BY (_ne.utm_medium, _ne.event_num) AS start_utm_medium,
      MIN_BY (_ne.utm_source, _ne.event_num) AS start_utm_source,
      MIN_BY (_ne.utm_term, _ne.event_num) AS start_utm_term,
      MIN_BY (_ne.referrer, _ne.event_num) AS start_referrer
    FROM
      ${ref("events_numbered")} AS _ne
    GROUP BY
      _ne.session_id
  )
SELECT
  ne.id,
  COALESCE(
    MIN_BY (ne3.start_url_path, ne3.event_num),
    MIN_BY (ne.url_path, ne.event_num)
  ) AS subsession_start_url_path,
  first_start_url_path.start_url_path AS session_start_url_path,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_utm_campaign, ne3.event_num)
    ELSE NULL
  END AS subsession_start_utm_campaign,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_utm_content, ne3.event_num)
    ELSE NULL
  END AS subsession_start_utm_content,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_utm_medium, ne3.event_num)
    ELSE NULL
  END AS subsession_start_utm_medium,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_utm_source, ne3.event_num)
    ELSE NULL
  END AS subsession_start_utm_source,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_utm_term, ne3.event_num)
    ELSE NULL
  END AS subsession_start_utm_term,
  CASE
    WHEN MIN_BY (ne3.start_referrer, ne3.event_num) != MIN_BY (ne3.last_referrer, ne3.event_num)
    OR MIN_BY (ne3.start_utm_campaign, ne3.event_num) != MIN_BY (ne3.last_utm_campaign, ne3.event_num)
    OR MIN_BY (ne3.start_utm_content, ne3.event_num) != MIN_BY (ne3.last_utm_content, ne3.event_num)
    OR MIN_BY (ne3.start_utm_medium, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_source, ne3.event_num) != MIN_BY (ne3.last_utm_medium, ne3.event_num)
    OR MIN_BY (ne3.start_utm_term, ne3.event_num) != MIN_BY (ne3.last_utm_term, ne3.event_num)
    OR (
      MIN_BY (ne3.last_utm_campaign, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_content, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_medium, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_source, ne3.event_num) IS NULL
      AND MIN_BY (ne3.last_utm_term, ne3.event_num) IS NULL
    ) THEN MIN_BY (ne3.start_referrer, ne3.event_num)
    ELSE NULL
  END AS subsession_start_referrer
FROM
  ${ref("events_numbered")} AS ne
  LEFT JOIN ${ref("events_start_urls")} AS ne3 ON ne.session_id = ne3.session_id
  AND ne.segment_id = ne3.segment_id
  AND ne.event_num >= ne3.event_num
  AND ne.utm_visit = ne3.utm_visit
  AND ne3.start_url_path IS NOT NULL
  LEFT JOIN session_ids_with_start_urls AS first_start_url_path ON ne.session_id = first_start_url_path.session_id
GROUP BY
  ne.id,
  ne.url_path,
  first_start_url_path.start_url_path
```
