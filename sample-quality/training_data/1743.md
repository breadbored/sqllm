--- USER ---

How can I decompose a user session into subsessions based on user changes and idle time gaps, and calculate timing metrics at both the session and subsession level?

--- AGENT ---

This query decomposes event sessions into subsessions using two criteria: changes in user_id within a session and idle time gaps exceeding 30 minutes. It progressively builds timing and segmentation metadata.

The logic flows as:

1. **User segment detection** — Identifies when the user_id changes within a session using a LAG window function, assigning each contiguous user block a `user_segment_id`.

2. **Idle gap detection** — Calculates the time elapsed between consecutive events (in minutes and seconds) using DATETIME_DIFF, then detects when gaps exceed 30 minutes to trigger subsession breaks. Also flags returns from "inactive" states (detected via the previous event being a ping/inactive action).

3. **Segment ID creation** — Concatenates a running count of 30+ minute gaps with the user_segment_id to create a composite `segment_id` that captures both user and idle-driven subsession boundaries.

4. **Per-segment timing recalculation** — Recomputes time deltas (minutes and seconds from last event) partitioned by both session_id and segment_id, so each new subsession resets its elapsed-time counter to 0 rather than preserving the large gap.

5. **Background return tracking** — Counts returns from inactive states at both session and subsession scope, useful for understanding user re-engagement patterns.

The output is a fully attributed event stream with segment IDs, timing columns at multiple levels, and return-from-background counters.

```sql
/**
* Subsessions
*
* This layer is used to get the amount of time between events, determine when a subsession starts, and assign a segment id to each subsession.
* Included is the ability to get the number of minutes between events, by segment.
**/
WITH
  sessions_with_users AS (
      SELECT
        e.session_id,
        e.user_id,
        e.created
      FROM
        ${ref("events_filter")} AS e
      WHERE
        e.user_id IS NOT NULL
  ),
  sessions_with_changing_user_ids AS (
    SELECT
      e.session_id,
      e.user_id,
      e.created,
      COALESCE(LAG(e.user_id) OVER (PARTITION BY e.session_id ORDER BY e.created ASC), e.user_id) as last_user_id
    FROM
      sessions_with_users as e
    -- QUALIFY
    --   e.user_id != last_user_id
  ),
  subsessions_with_segments AS (
    SELECT
      e.session_id,
      e.created,
      SUM(
        CASE
          WHEN e.user_id != e.last_user_id THEN 1
          ELSE 0
        END
      ) OVER (
        PARTITION BY
          e.session_id
        ORDER BY
          e.created ASC
      ) AS user_segment_id
    FROM
      sessions_with_changing_user_ids as e
  ),
  user_segments AS (
    SELECT
      e.id,
      COALESCE(MAX(sws.user_segment_id), 0) as user_segment_id
    FROM
      ${ref("events_filter")} as e
    LEFT JOIN
      subsessions_with_segments as sws
    ON
      e.session_id = sws.session_id
    AND
      e.created >= sws.created
    GROUP BY
      e.id
  ),
  user_segment_events AS (
    SELECT
      e.*,
      us.user_segment_id
    FROM
      ${ref("events_filter")} as e
    JOIN
      user_segments as us
    ON
      us.id = e.id
  ),
  -- Get time between events
  idle_counter AS (
    SELECT
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id
            ORDER BY
              e.created ASC
          ),
          MINUTE
        ),
        0
      ) AS minutes_from_last_event_with_idle,
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id
            ORDER BY
              e.created ASC
          ),
          SECOND
        ),
        0
      ) AS seconds_from_last_event_with_idle,
      COALESCE(
        LAG(e.name) OVER (
          PARTITION BY
            e.session_id
          ORDER BY
            e.created ASC
        ) = 'ping' 
      AND 
        LAG(e.action) OVER (
          PARTITION BY
            e.session_id
          ORDER BY
            e.created ASC
        ) = 'inactive' 
      AND
        (
          e.name != 'ping'
        OR
          e.action != 'inactive'
        ),
        false
      ) as back_from_inactive,
      *
    FROM
      user_segment_events AS e
    ORDER BY
      e.created ASC
  ),
  -- Get time between events
  timed_events AS (
    SELECT
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id
            ORDER BY
              e.created ASC
          ),
          MINUTE
        ),
        0
      ) AS minutes_from_last_event,
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id
            ORDER BY
              e.created ASC
          ),
          SECOND
        ),
        0
      ) AS seconds_from_last_event,
      *
    FROM
      idle_counter AS e
    WHERE
      NOT (e.name = 'ping' AND e.action = 'inactive')
    ORDER BY
      e.created ASC
  ),
  -- Iterate the "sub-session" segment every time a gap larger than 30 minutes comes up
  sectioned_events AS (
    SELECT
      CONCAT(
        SUM(
          CASE
            WHEN te.minutes_from_last_event > 30 THEN 1
            ELSE 0
          END
        ) OVER (
          PARTITION BY
            te.session_id
          ORDER BY
            te.created ASC
        ),
        '-',
        te.user_segment_id
      ) AS segment_id,
      SUM(
        CASE
          WHEN te.back_from_inactive AND te.minutes_from_last_event_with_idle < 30 THEN 1
          ELSE 0
        END
      ) OVER (
        PARTITION BY
          te.session_id
        ORDER BY
          te.created ASC
      ) AS returned_from_background_count,
      *
    FROM
      timed_events AS te
    ORDER BY
      te.created ASC
  ),
  -- This is used to get the number of minutes between events, by segment, meaning new segments will start with 0 instead of >30 minutes
  timed_segments_events AS (
    SELECT
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id,
              e.segment_id
            ORDER BY
              e.created ASC
          ),
          MINUTE
        ),
        0
      ) AS minutes_from_last,
      COALESCE(
        DATETIME_DIFF(
          e.created,
          LAG(e.created) OVER (
            PARTITION BY
              e.session_id,
              e.segment_id
            ORDER BY
              e.created ASC
          ),
          SECOND
        ),
        0
      ) AS seconds_from_last,
      SUM(
        CASE
          WHEN e.back_from_inactive AND e.minutes_from_last_event_with_idle < 30 THEN 1
          ELSE 0
        END
      ) OVER (
        PARTITION BY
          e.session_id,
          e.segment_id
        ORDER BY
          e.created ASC
      ) AS returned_from_background_subsession_count,
      *
    FROM
      sectioned_events AS e
    ORDER BY
      e.created ASC
  )
SELECT
  *
FROM
  timed_segments_events
```
