## Step 7: Compose the Final Query

```sql
WITH
  min_months AS (
    -- Get the min month per user per course
  SELECT
    user_id,
    course_id,
    DATE_TRUNC(progress_date, MONTH) AS min_month
  FROM
    ${ref("agg_daily_user_course_progress")}
  GROUP BY
    user_id,
    course_id,
    min_month),
  latest_day_in_min_month AS (
    -- Get the max day within that min month per user per course
  SELECT
    a.user_id,
    a.course_id,
    a.progress_date,
    a.total_progress_seconds,
    a.total_time_on_course,
    a.total_course_progress_percentage,
    ROW_NUMBER() OVER (PARTITION BY a.user_id, a.course_id ORDER BY a.progress_date DESC ) AS rn
  FROM
    ${ref("agg_daily_user_course_progress")} a
  JOIN
    min_months b
  ON
    a.user_id = b.user_id
    AND a.course_id = b.course_id
    AND DATE_TRUNC(a.progress_date, MONTH) = b.min_month),
  final_progress AS (
  SELECT
    user_id,
    course_id,
    progress_date,
    MAX(total_progress_seconds) AS total_progress_seconds,
    MAX(total_time_on_course) AS total_time_on_course,
    MAX(total_course_progress_percentage) AS total_course_progress_percentage
  FROM
    latest_day_in_min_month
  WHERE
    rn = 1
  GROUP BY
    user_id,
    course_id,
    progress_date)
SELECT
  fp.user_id,
  fp.course_id,
  c.container_id,
  fp.progress_date,
  CASE
    WHEN CAST(MAX(fp.total_course_progress_percentage) AS FLOAT64) = 1 AND uc.status = "completed" THEN "completed"
    WHEN CAST(MAX(fp.total_course_progress_percentage) AS FLOAT64) >= 0.95 THEN "100%"
    WHEN CAST(MAX(fp.total_course_progress_percentage) AS FLOAT64) < 0.95 THEN "in_progress"
    WHEN MAX(fp.total_course_progress_percentage) IS NULL THEN "not_started"
END
  AS derived_course_status,
  MAX(fp.total_progress_seconds) AS total_progress_seconds,
  MAX(fp.total_time_on_course) AS total_time_on_course,
  MAX(fp.total_course_progress_percentage) AS total_course_progress_percentage,
  ARRAY_AGG(DISTINCT clients.id) AS client_id,
  ARRAY_AGG(DISTINCT clients.name) AS client_name
FROM
  final_progress fp
JOIN
  ${ref("stg_user_courses")} uc
ON
  uc.user_id = fp.user_id
  AND uc.course_id = fp.course_id
JOIN
  ${ref("stg_courses")} c
ON
  c.id = uc.course_id
JOIN
  ${ref("stg_clients_course_containers")} ccc
ON
  ccc.course_container_id = c.container_id
JOIN
  ${ref("stg_clients")} clients
ON
  clients.id = ccc.client_id
GROUP BY
  fp.user_id,
  fp.course_id,
  c.container_id,
  fp.progress_date,
  uc.status
```
