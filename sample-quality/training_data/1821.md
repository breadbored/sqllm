--- USER ---

Generate a BigQuery query that enriches user session events with engagement metrics, content metadata, authentication details, and associated client information to create a comprehensive analytics fact table.

--- AGENT ---

This query constructs a denormalized session-level fact table by joining enriched event data with multiple lookup tables.

**Data flow:**

The query starts with `events_acquisition`, which groups raw events by session and sub_session, extracting a landing page path, engagement metrics (time on page, scroll depth, click counts), and resource IDs (course_id, container_id, badge_id) parsed from URL patterns using regex. UTM parameters and referrer are also extracted and normalized (underscores/plus signs/encoded spaces converted to spaces, lowercased).

`subsession_data` calculates per-session rollups: page view count, session duration, click count, and an "engaged" flag (TRUE unless the session was a bounce: <30 seconds, single page, zero clicks).

Three metadata CTEs (`course_data`, `badge_data`, `channel_data`) perform lookups to resolve what content a user landed on: course name and client associations, badge name and container/client associations, or channel name and container associations.

`auth_data` determines whether the session contained a login or register event, and classifies the auth action type.

`user_course_data` pulls the user's course progress for any courses started on that date, ranked by start time, keeping only the first course per user per day.

The final SELECT performs a series of LEFT JOINs to combine all this data into a single row per session. The join to clients is many-to-many (allowing multiple clients per session based on course/badge/channel), so client IDs and names are aggregated into arrays using `ARRAY_CONCAT` (defaulting to [1] / ['Data Protocol'] if no clients are found).

**The query:**

```sql
WITH
  events_acquisition AS (
  SELECT
    e.session_id,
    e.sub_session_id,
    MAX(e.user_id) AS user_id,
    MAX(e.current_user_id) AS current_user_id,
    MIN_BY(e.url_path, e.created) AS landing_path,
    DATETIME_DIFF(MAX(e.created), MIN(e.created), SECOND) AS time_on_page,
    MAX(e.scroll_value) AS max_scroll_depth,
    MAX(e.scroll_num) AS scroll_events_count,
    COUNTIF(e.type = "click"
      AND e.name != "any_click") AS landing_page_click_count,
    SAFE_CAST(REGEXP_EXTRACT( MIN_BY(e.url_path, e.created), r'/(?:courses|short-codes)/([^/]+)') AS int64) AS course_id,
    SAFE_CAST(REGEXP_EXTRACT( MIN_BY(e.url_path, e.created), r'/container/([^/]+)') AS int64) AS container_id,
    SAFE_CAST(REGEXP_EXTRACT( MIN_BY(e.url_path, e.created), r'/badges/([^/]+)') AS int64) AS badge_id,
    MIN_BY(e.referrer, e.created) AS referrer,
    LOWER(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE( REGEXP_EXTRACT(MIN_BY(url, created), r'utm_source=([^&]*)'), r'_', ' '), r'\+', ' '), r'%20', ' '), r'-', ' ')) AS utm_source,
    LOWER(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE( REGEXP_EXTRACT(MIN_BY(url, created), r'utm_medium=([^&]*)'), r'_', ' '), r'\+', ' '), r'%20', ' '), r'-', ' ')) AS utm_medium,
    LOWER(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE( REGEXP_EXTRACT(MIN_BY(url, created), r'utm_campaign=([^&]*)'), r'_', ' '), r'\+', ' '), r'%20', ' '), r'-', ' ')) AS utm_campaign,
    LOWER(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE( REGEXP_EXTRACT(MIN_BY(url, created), r'utm_content=([^&]*)'), r'_', ' '), r'\+', ' '), r'%20', ' '), r'-', ' ')) AS utm_content,
    LOWER(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE(REGEXP_REPLACE( REGEXP_EXTRACT(MIN_BY(url, created), r'utm_term=([^&]*)'), r'_', ' '), r'\+', ' '), r'%20', ' '), r'-', ' ')) AS utm_term,
    MAX(e.landed_from) as landed_from,
    MIN_BY(e.page_visit, e.created) AS page_visit,
    MIN(e.created) AS created
  FROM
    ${ref("stg_events_app_speedy")} e
  GROUP BY
    e.session_id,
    e.sub_session_id),
  subsession_data AS (
  SELECT
    sub_session_id,
    COUNT(DISTINCT url_path) AS page_views,
    DATETIME_DIFF(MAX(created), MIN(created), SECOND) AS session_duration,
    COUNTIF(type = "click") AS session_click_count,
    CASE
      WHEN DATETIME_DIFF(MAX(created), MIN(created), SECOND) < 30 AND COUNT(DISTINCT url_path) = 1 AND COUNTIF(type = "click") = 0 THEN FALSE
    -- Bounce criteria
      ELSE TRUE
    -- Engaged if not a bounce
  END
    AS engaged,
  FROM
    ${ref("stg_events_app_speedy")}
  GROUP BY
    sub_session_id),
  went_to_docs AS (
  SELECT
    e.sub_session_id,
    MIN_BY(e.app, e.created) = 'docs' AS started_at_docs,
    MIN(e.app) = 'docs' AS went_to_docs
  FROM
    ${ref("stg_events_speedy")} AS e
  GROUP BY
    e.sub_session_id),
  course_data AS (
  SELECT
    MAX_BY(c.name, c.id) AS course_name,
    ARRAY_AGG(c.id) AS course_id,
    c.container_id AS container_id,
    ARRAY_AGG(DISTINCT ccc.client_id) AS client_id,
    ARRAY_AGG(DISTINCT clients.name) AS client_name
  FROM
    ${ref("stg_clients_course_containers")} ccc
  JOIN
    ${ref("stg_clients")} clients
  ON
    clients.id = ccc.client_id
  JOIN
    ${ref("stg_courses")} c
  ON
    c.container_id = ccc.course_container_id
  GROUP BY
    c.container_id),
  badge_data AS (
  SELECT
    lp.id AS badge_id,
    lp.name AS badge_name,
    ARRAY_AGG(DISTINCT lpcc.course_container_id) AS container_ids,
    ARRAY_AGG(DISTINCT ccc.client_id) AS client_id,
    ARRAY_AGG(DISTINCT clients.name) AS client_name
  FROM
    ${ref("stg_learning_paths")} lp
  JOIN
    ${ref("learning_path_course_containers")} lpcc
  ON
    lpcc.learning_path_id = lp.id
  JOIN
    ${ref("clients_course_containers")} ccc
  ON
    ccc.course_container_id = lpcc.course_container_id
  JOIN
    ${ref("stg_clients")} clients
  ON
    clients.id = ccc.client_id
  GROUP BY
    lp.id,
    lp.name),
  channel_data AS (
  SELECT
    cats.link,
    cats.name AS client_name,
    ARRAY_AGG(ccc.course_container_id) AS channel_container_ids
  FROM
    ${ref("stg_categories")} cats
  JOIN
    ${ref("stg_clients")} clients
  ON
    LOWER(clients.name) = cats.name
  JOIN
    ${ref("stg_clients_course_containers")} ccc
  ON
    ccc.client_id = clients.id
  WHERE
    link IS NOT NULL
  GROUP BY
    cats.link,
    cats.name),
  auth_data AS (
  SELECT
    e.session_id,
    e.sub_session_id,
    MAX_BY(e.type, e.created) AS auth_type,
    CASE
      WHEN MAX_BY(e.action, e.created) LIKE '%register%' THEN 'register'
      WHEN MAX_BY(e.type, e.created) = 'metaquest'
    AND ABS(TIMESTAMP_DIFF(MIN(e.created), MIN(u.created), SECOND)) <= 5 THEN 'register'
      ELSE 'login'
  END
    AS auth_action
  FROM
    ${ref("stg_events_app_speedy")} AS e
  JOIN
    ${ref("stg_users")} AS u
  ON
    u.id = e.user_id
  WHERE
    e.name = "login"
  GROUP BY
    e.session_id,
    e.sub_session_id),
  user_course_data AS (
  SELECT
    *,
    ROW_NUMBER() OVER (PARTITION BY user_id, DATE (course_started)
    ORDER BY
      course_started ASC ) AS rn
  FROM (
    SELECT
      c.user_id,
      c.course_name,
      c.container_id,
      c.course_id,
      c.course_status,
      c.course_progress_percentage,
      c.course_started,
      ARRAY_AGG(DISTINCT CCC.client_id IGNORE NULLS) AS client_ids
    FROM
      ${ref("reporting_user_courses")} c
    JOIN
      ${ref("stg_clients_course_containers")} ccc
    ON
      ccc.course_container_id = c.container_id
    GROUP BY
      user_id,
      course_name,
      container_id,
      course_id,
      course_status,
      course_progress_percentage,
      course_started))
SELECT
  e.session_id,
  e.sub_session_id,
  e.user_id,
  e.current_user_id,
  e.landing_path,
  COALESCE(c.course_name, bd.badge_name, CONCAT(INITCAP(cd.client_name), ' Channel')) AS name,
  CASE
    WHEN landing_path LIKE '/courses/%' OR landing_path LIKE '/short-codes/%' OR landing_path LIKE '/container/%' THEN 'Course'
    WHEN landing_path LIKE '/channels/%' THEN 'Channel'
    WHEN landing_path LIKE '/badges/%' THEN 'Badge'
END
  AS type,
  e.time_on_page,
  e.landing_page_click_count,
  ad.auth_action,
  ad.auth_type,
  COALESCE(MAX(wtd.started_at_docs), FALSE) AS started_at_docs,
  COALESCE(MAX(wtd.went_to_docs), FALSE) AS went_to_docs,
  sd.page_views,
  sd.session_duration,
  sd.session_click_count,
  sd.engaged,
  ucd.course_name,
  ucd.container_id,
  ucd.course_id,
  ucd.course_status,
  ucd.course_progress_percentage,
  ucd.course_started,
  e.utm_source,
  e.utm_medium,
  e.utm_campaign,
  e.utm_content,
  e.referrer,
  e.landed_from,
  e.created,
  COALESCE(ARRAY_CONCAT(ARRAY_AGG(clients.id IGNORE NULLS), [1]), [1]) AS client_id,
  COALESCE(ARRAY_CONCAT(ARRAY_AGG(clients.name IGNORE NULLS), ['Data Protocol']), ['Data Protocol']) AS client_name
FROM
  events_acquisition e
LEFT JOIN
  course_data c
ON
  e.container_id = c.container_id
  OR e.course_id IN UNNEST(c.course_id)
LEFT JOIN
  badge_data bd
ON
  bd.badge_id = e.badge_id
LEFT JOIN
  channel_data cd
ON
  cd.link =
  CASE
    WHEN e.landing_path = "/channels/meta-quest" THEN "/channels/meta-horizon"
    ELSE e.landing_path
END
LEFT JOIN
  auth_data ad
ON
  ad.sub_session_id = e.sub_session_id
LEFT JOIN
  ${ref("stg_clients")} clients
ON
  (LOWER(clients.name) = cd.client_name
    OR clients.id IN UNNEST(bd.client_id)
    OR clients.id IN UNNEST(c.client_id))
  AND NOT clients.id = 1
LEFT JOIN
  subsession_data AS sd
ON
  sd.sub_session_id = e.sub_session_id
LEFT JOIN
  user_course_data ucd
ON
  e.user_id = ucd.user_id
  AND DATE(e.created) = DATE(ucd.course_started)
  AND (c.container_id = ucd.container_id
    OR ucd.container_id IN UNNEST(bd.container_ids)
    OR ucd.container_id IN UNNEST(cd.channel_container_ids))
  AND ucd.rn = 1
  -- Ensures only the first course per user per day
LEFT JOIN
  went_to_docs AS wtd
ON
  wtd.sub_session_id = e.sub_session_id
GROUP BY
  e.session_id,
  e.sub_session_id,
  e.user_id,
  e.current_user_id,
  e.landing_path,
  e.time_on_page,
  e.landing_page_click_count,
  COALESCE(c.course_name, bd.badge_name, CONCAT(INITCAP(cd.client_name), ' Channel')),
  ad.auth_action,
  ad.auth_type,
  sd.page_views,
  sd.session_duration,
  sd.session_click_count,
  sd.engaged,
  ucd.course_name,
  ucd.container_id,
  ucd.course_id,
  ucd.course_status,
  ucd.course_progress_percentage,
  ucd.course_started,
  e.utm_source,
  e.utm_medium,
  e.utm_campaign,
  e.utm_content,
  e.referrer,
  e.landed_from,
  e.created
```
