--- USER ---

Provide me a BigQuery SQL query that enriches user course progress records with detailed progress metrics for the first and second lessons in each course.

--- AGENT ---

This query enriches base user course progress records with lesson-specific progress metrics for the first two lessons in each course.

The query works in three logical layers:

1. **Lesson Identification CTEs**: The `rankedlessons` CTE ranks lessons within each course by ID, and `secondlesson` filters to isolate just the second lesson (rank = 2) for each course. The first lesson is identified inline using a MIN aggregation.

2. **First Lesson Enrichment**: An INNER JOIN retrieves the user's progress on the first lesson (identified by `MIN(id)` within the course). For each user-course pair, it calculates:
   - The user's maximum progress in seconds on that lesson
   - The lesson's duration (from video metadata)
   - Progress percentage, normalized to a maximum of 1.0 using `LEAST`

3. **Second Lesson Enrichment**: A LEFT JOIN applies the same pattern to the second lesson (identified via the `secondlesson` CTE). Since not all users may have progress on the second lesson, this join is optional (LEFT rather than INNER).

Both lesson joins include a nested `LEFT JOIN` to a completed-lessons aggregation, though this intermediate result is not used in the final output columns — it appears to be unused context from the query's source.

The final output combines all user course progress columns with seven new columns: first/second lesson duration, user progress in seconds, and progress percentage for each.

```sql
WITH
rankedlessons AS (
  SELECT
    course_id,
    id,
    row_number() OVER (PARTITION BY course_id ORDER BY id) AS rn
  FROM
    ${ref("stg_course_lessons")}
),

secondlesson AS (
  SELECT
    rl.course_id,
    rl.id
  FROM
    rankedlessons AS rl
  WHERE
    rl.rn = 2
)

SELECT
  ucp.*,
  uflp.first_lesson_length,
  uflp.user_first_lesson_progress,
  uflp.first_lesson_progress_percentage,
  uslp.second_lesson_length,
  uslp.user_second_lesson_progress,
  uslp.second_lesson_progress_percentage
FROM
  ${ref("agg_user_course_progress")} AS ucp
  -- first lesson
INNER JOIN (
  SELECT
    uclp.user_id,
    uclp.course_id,
    uclp.lesson_id,
    max(uclp.progress_seconds)                                                   AS user_first_lesson_progress,
    fld.first_lesson_length,
    least(round(((max(uclp.progress_seconds) / fld.first_lesson_length)), 2), 1) AS first_lesson_progress_percentage
  FROM
    ${ref("stg_user_course_lesson_progress")} AS uclp
  INNER JOIN (
    SELECT
      cl.course_id,
      cl.lesson_id AS first_lesson_id,
      v.duration   AS first_lesson_length
    FROM
      ${ref("stg_course_lessons")} AS cl
    INNER JOIN (
      SELECT
        course_id,
        min(id) AS id
      FROM
        ${ref("stg_course_lessons")}
      GROUP BY
        course_id
    ) AS fl
      ON
        cl.id = fl.id
        AND cl.course_id = fl.course_id
    INNER JOIN
      ${ref("stg_lessons")} AS l
      ON
        cl.lesson_id = l.id
    INNER JOIN
      ${ref("stg_videos")} AS v
      ON
        l.video_id = v.id
  ) AS fld
    ON
      uclp.course_id = fld.course_id
      AND uclp.lesson_id = fld.first_lesson_id
  LEFT JOIN (
    SELECT
      ucp.user_id,
      ucp.course_id,
      count(DISTINCT ufcl.lesson_id) AS completed_lessons
    FROM
      ${ref("agg_user_course_progress")} AS ucp

    LEFT JOIN
      ${ref("stg_user_finished_course_lessons")} AS ufcl
      ON
        ucp.course_id = ufcl.course_id
        AND ucp.user_id = ufcl.user_id
    GROUP BY
      ucp.user_id,
      ucp.course_id
  ) AS ucl
    ON
      uclp.user_id = ucl.user_id
      AND uclp.course_id = ucl.course_id
  GROUP BY
    uclp.user_id,
    uclp.course_id,
    lesson_id,
    fld.first_lesson_length
) AS uflp
  ON
    ucp.user_id = uflp.user_id
    AND ucp.course_id = uflp.course_id
-- second lesson
LEFT JOIN (
  SELECT
    uclp.user_id,
    uclp.course_id,
    uclp.lesson_id,
    max(uclp.progress_seconds)                                                    AS user_second_lesson_progress,
    sld.second_lesson_length,
    least(round(((max(uclp.progress_seconds) / sld.second_lesson_length)), 2), 1) AS second_lesson_progress_percentage
  FROM
    ${ref("stg_user_course_lesson_progress")} AS uclp
  INNER JOIN (
    SELECT
      cl.course_id,
      cl.lesson_id AS second_lesson_id,
      v.duration   AS second_lesson_length
    FROM
      ${ref("stg_course_lessons")} AS cl
      -- get second lesson
    INNER JOIN
      secondlesson AS sl
      ON
        cl.id = sl.id
        AND cl.course_id = sl.course_id
    INNER JOIN
      ${ref("stg_lessons")} AS l
      ON
        cl.lesson_id = l.id
    INNER JOIN
      ${ref("stg_videos")} AS v
      ON
        l.video_id = v.id
  ) AS sld
    ON
      uclp.course_id = sld.course_id
      AND uclp.lesson_id = sld.second_lesson_id
  LEFT JOIN (
    SELECT
      ucp.user_id,
      ucp.course_id,
      count(DISTINCT ufcl.lesson_id) AS completed_lessons
    FROM
      ${ref("agg_user_course_progress")} AS ucp

    LEFT JOIN
      ${ref("stg_user_finished_course_lessons")} AS ufcl
      ON
        ucp.course_id = ufcl.course_id
        AND ucp.user_id = ufcl.user_id
    GROUP BY
      ucp.user_id,
      ucp.course_id
  ) AS ucl
    ON
      uclp.user_id = ucl.user_id
      AND uclp.course_id = ucl.course_id
  GROUP BY
    uclp.user_id,
    uclp.course_id,
    lesson_id,
    sld.second_lesson_length
) AS uslp
  ON
    ucp.user_id = uslp.user_id
    AND ucp.course_id = uslp.course_id
```
