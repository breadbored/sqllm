## Question
Provide me with an example BigQuery SQL query that analyzes multi-level referral networks and identifies the role (sharer or sharee) of each user within those chains.

--- AGENT ---

This query analyzes a multi-level referral network structure (up to 4 levels deep) in an event stream. It identifies chains of referrals where sessions refer other sessions, then categorizes each participant as either a "sharer" (someone who referred others) or "sharee" (someone who was referred).

The logic works as follows:

1. **Identify referral relationships**: The `referrals` CTE extracts sessions with referral data from the events table, filtering for apps 'app' or 'docs'.

2. **Build referral chains**: The `session_referrals` CTE performs recursive self-joins on the referrals CTE, connecting each referrer to their referees up to 4 levels deep via the `referred_by` foreign key. At each join level, it also retrieves the `sub_session_id` for behavioral tracking.

3. **Aggregate into level-based arrays**: The `session_level_users` CTE collects all `sub_session_id` values at each level into separate arrays (_1st_level_users through _4th_level_users).

4. **Unwrap and tag**: The `unwrapped_shared_sessions` CTE uses 5 UNION ALL branches to unwrap each array level and assign roles. Users at level 1 are tagged as "sharer"; users at level 2 are "sharee" (or "sharer" if they have referees at level 3); users at level 3 are "sharee" (or "sharer" if they have referees at level 4); users at level 4 are "sharee".

5. **Attach behavioral metadata**: The final SELECT joins the unwrapped session data back to the event stream, grouping by sub_session_id and tag to retrieve the first app and URL path (using MIN_BY to select by earliest timestamp) and the session start time.

The output includes two variants via UNION ALL: one with simple role names and one with role concatenated to level number (e.g., 'sharer_2', 'sharee_3').

```sql
WITH
when_referred AS (
  SELECT
    e.session_id as session_id,
    MIN(e.created) session_started
  FROM
    ${ref("base_buckets_events_app_speedy")} as e
  WHERE
    e.referred_by IS NOT NULL
  GROUP BY
    e.session_id
),
referrals AS (
  SELECT
    e.session_id as session_id,
    MAX(e.referred_by) as referred_by,
    COALESCE(MIN(wr.session_started), MIN(e.created)) as session_started
  FROM
    ${ref("base_buckets_events_app_speedy")} as e
  LEFT JOIN
    when_referred as wr
  ON
    wr.session_id = e.session_id
  WHERE
    (e.app = 'app' OR e.app = 'docs')
  GROUP BY
    e.session_id
),
session_users AS (
  SELECT
    e.session_id as session_id,
    MAX(e.sub_session_id) as sub_session_id
  FROM
    ${ref("base_buckets_events_app_speedy")} as e
  GROUP BY
    e.session_id
),
session_referrals AS (
  SELECT
    r.session_id as session_id_1,
    MIN(r.session_started) as session_started_1,
    su.sub_session_id as sub_session_id_1,
    r2.session_id as session_id_2,
    MIN(r2.session_started) as session_started_2,
    su2.sub_session_id as sub_session_id_2,
    r3.session_id as session_id_3,
    MIN(r3.session_started) as session_started_3,
    su3.sub_session_id as sub_session_id_3,
    r4.session_id as session_id_4,
    MIN(r4.session_started) as session_started_4,
    su4.sub_session_id as sub_session_id_4,
  FROM
    referrals as r
  JOIN
    session_users as su
  ON
    su.session_id = r.session_id
  JOIN
    referrals as r2
  ON
    r.session_id = r2.referred_by
  AND
    r.session_id != r2.session_id
  JOIN
    session_users as su2
  ON
    su2.session_id = r2.session_id
  LEFT JOIN
    referrals as r3
  ON
    r2.session_id = r3.referred_by
  AND
    r2.session_id != r3.session_id
  LEFT JOIN
    session_users as su3
  ON
    su3.session_id = r3.session_id
  LEFT JOIN
    referrals as r4
  ON
    r3.session_id = r4.referred_by
  AND
    r3.session_id != r4.session_id
  LEFT JOIN
    session_users as su4
  ON
    su4.session_id = r4.session_id
  GROUP BY
    session_id_4,
    sub_session_id_4,
    session_id_3,
    sub_session_id_3,
    session_id_2,
    sub_session_id_2,
    session_id_1,
    sub_session_id_1
  ORDER BY
    sub_session_id_4 DESC,
    session_id_4 DESC,
    sub_session_id_3 DESC,
    session_id_3 DESC,
    sub_session_id_2 DESC,
    session_id_2 DESC,
    sub_session_id_1 DESC,
    session_id_1 DESC
),
session_level_users AS (
  SELECT
    ARRAY_AGG(
      sf.sub_session_id_1
      IGNORE NULLS
    ) as _1st_level_users,
    ARRAY_AGG(
      sf.sub_session_id_2
      IGNORE NULLS
    ) as _2nd_level_users,
    ARRAY_AGG(
      sf.sub_session_id_3
      IGNORE NULLS
    ) as _3rd_level_users,
    ARRAY_AGG(
      sf.sub_session_id_4
      IGNORE NULLS
    ) as _4th_level_users
  FROM
    session_referrals as sf
),
unwrapped_shared_sessions AS (
  SELECT
    'sharer' as tag,
    1 as level,
    _1lu AS sub_session_id,
    MIN(sr.session_started_1) as session_started
  FROM
    session_level_users as slu, slu._1st_level_users as _1lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_1 = _1lu
  GROUP BY
    sub_session_id

  UNION ALL

  SELECT
    'sharee' as tag,
    2 as level,
    _2lu AS sub_session_id,
    MIN(sr.session_started_2) as session_started
  FROM
    session_level_users as slu, slu._2nd_level_users as _2lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_2 = _2lu
  GROUP BY
    sub_session_id

  UNION ALL

  SELECT
    'sharer' as tag,
    2 as level,
    _2lu AS sub_session_id,
    MIN(sr.session_started_2) as session_started
  FROM
    session_level_users as slu, slu._2nd_level_users as _2lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_2 = _2lu
  WHERE
    ARRAY_LENGTH(slu._3rd_level_users) > 0
  GROUP BY
    sub_session_id

  UNION ALL

  SELECT
    'sharee' as tag,
    3 as level,
    _3lu AS sub_session_id,
    MIN(sr.session_started_3) as session_started
  FROM
    session_level_users as slu, slu._3rd_level_users as _3lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_3 = _3lu
  GROUP BY
    sub_session_id

  UNION ALL

  SELECT
    'sharer' as tag,
    3 as level,
    _3lu AS sub_session_id,
    MIN(sr.session_started_3) as session_started
  FROM
    session_level_users as slu, slu._3rd_level_users as _3lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_3 = _3lu
  WHERE
    ARRAY_LENGTH(slu._4th_level_users) > 0
  GROUP BY
    sub_session_id

  UNION ALL

  SELECT
    'sharee' as tag,
    4 as level,
    _4lu AS sub_session_id,
    MIN(sr.session_started_4) as session_started
  FROM
    session_level_users as slu, slu._4th_level_users as _4lu
  JOIN
    session_referrals as sr
  ON
    sr.sub_session_id_4 = _4lu
  GROUP BY
    sub_session_id
)
SELECT
  uss.sub_session_id as sub_session_id,
  uss.tag as tag,
  'behavior' as class,
  MIN_BY(e.app, e.created) as app,
  MIN_BY(e.url_path, e.created) as url_path,
  MIN(e.created) as dt
FROM
  unwrapped_shared_sessions as uss
JOIN
  ${ref("base_buckets_events_app_speedy")} as e
ON
  uss.sub_session_id = e.sub_session_id
GROUP BY
  sub_session_id,
  tag

UNION ALL

SELECT
  uss.sub_session_id as sub_session_id,
  CONCAT(uss.tag, '_', uss.level) as tag,
  'behavior' as class,
  MIN_BY(e.app, e.created) as app,
  MIN_BY(e.url_path, e.created) as url_path,
  MIN(e.created) as dt
FROM
  unwrapped_shared_sessions as uss
JOIN
  ${ref("base_buckets_events_app_speedy")} as e
ON
  uss.sub_session_id = e.sub_session_id
GROUP BY
  sub_session_id,
  tag
```
